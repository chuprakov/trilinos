INCLUDE(PackageAddExecutable)

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}"
)

IF(TPL_ENABLE_MATLAB)
ADD_OPTION_AND_DEFINE(${PACKAGE_NAME}_ENABLE_MATLAB
  HAVE_SEACAS_MATLAB
  "Enable Matlab Functionality."
  OFF )
INCLUDE_DIRECTORIES(${MATLAB_INCLUDE_DIRS})
LINK_DIRECTORIES(${MATLAB_LIBRARY_DIRS})
SET(DEPMATLIB mx mex mat)
ADD_DEFINITIONS(-DHAS_MATLAB)
ENDIF()

SET(CODE exo2mat)
SET(HEADERS "")
SET(SOURCES add_to_log.c  exo2mat.c)

PACKAGE_ADD_EXECUTABLE(${CODE} NOEXEPREFIX NOEXESUFFIX SOURCES ${SOURCES} DEPLIBS exodus ${DEPMATLIB} COMM serial)
install_executable(${CODE})
