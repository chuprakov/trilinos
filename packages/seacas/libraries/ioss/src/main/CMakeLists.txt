INCLUDE(PackageLibraryMacros)
INCLUDE(PackageAddExecutable)
INCLUDE(CheckIncludeFile)

SET(HEADERS "")
SET(SOURCES "")

SET_AND_INC_DIRS(DIR ${CMAKE_CURRENT_SOURCE_DIR})
APPEND_GLOB(HEADERS ${DIR}/*.h)
APPEND_GLOB(SOURCES ${DIR}/*.C)

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}/../"
  "${CMAKE_CURRENT_BINARY_DIR}/../"
)

ASSERT_DEFINED(${PACKAGE_NAME}_ENABLE_XDMF)
IF (NOT ${PACKAGE_NAME}_ENABLE_XDMF)
  ADD_DEFINITIONS(-DNO_XDMF_SUPPORT)
ENDIF()


PACKAGE_ADD_EXECUTABLE(cth_pressure_map SOURCES cth_pressure_map.C DEPLIBS Ionit)
PACKAGE_ADD_EXECUTABLE(io_shell         SOURCES io_shell.C DEPLIBS Ionit)
PACKAGE_ADD_EXECUTABLE(shell_to_hex     SOURCES shell_to_hex.C DEPLIBS Ionit)
PACKAGE_ADD_EXECUTABLE(io_info          SOURCES io_info.C volume.C DEPLIBS Ionit)
