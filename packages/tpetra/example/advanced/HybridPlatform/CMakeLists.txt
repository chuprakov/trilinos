    COMM mpi # serial

  TRIBITS_ADD_EXECUTABLE_AND_TEST(
    HybridPlatformTutorial
    SOURCES
      HybridPlatformTutorial
    # ARGS
    COMM mpi # serial
    STANDARD_PASS_OUTPUT
    )
ENDIF()
