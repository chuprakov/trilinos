# Create the PyTrilinos_PACKAGES string
SET(${PACKAGE_NAME}_PACKAGES "")
FOREACH(MODULE ${${PACKAGE_NAME}_MODULES})
  SET(${PACKAGE_NAME}_PACKAGES "${${PACKAGE_NAME}_PACKAGES}    ${MODULE}\n")
ENDFOREACH(MODULE ${${PACKAGE_NAME}_MODULES})

# Create the PyTrilinos_ALL string
SET(${PACKAGE_NAME}_ALL "__all__ = [")
FOREACH(MODULE ${${PACKAGE_NAME}_MODULES})
  SET(${PACKAGE_NAME}_ALL "${${PACKAGE_NAME}_ALL}'${MODULE}',\n          ")
ENDFOREACH(MODULE ${${PACKAGE_NAME}_MODULES})
SET(${PACKAGE_NAME}_ALL "${${PACKAGE_NAME}_ALL}]")

# Create the __init__.py file
CONFIGURE_FILE(${CMAKE_CURRENT_SOURCE_DIR}/__init__.src
  ${CMAKE_CURRENT_BINARY_DIR}/__init__.py)
