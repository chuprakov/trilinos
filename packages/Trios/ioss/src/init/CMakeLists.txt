INCLUDE(PackageLibraryMacros)
INCLUDE(CheckIncludeFile)

SET(HEADERS "")
SET(SOURCES "")

SET_AND_INC_DIRS(DIR ${CMAKE_CURRENT_SOURCE_DIR})
APPEND_GLOB(HEADERS ${DIR}/*.h)
APPEND_GLOB(SOURCES ${DIR}/*.C)

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}/../"
  "${CMAKE_CURRENT_BINARY_DIR}/../"
)

ASSERT_DEFINED(${PACKAGE_NAME}_ENABLE_USE_WITH_SIERRA)
IF (NOT ${PACKAGE_NAME}_ENABLE_USE_WITH_SIERRA)
ADD_DEFINITIONS(-DIOSS_STANDALONE)
ENDIF()

PACKAGE_ADD_LIBRARY(
	Ionit
	HEADERS	${HEADERS}
	SOURCES ${SOURCES}
	DEPLIBS Ioss Ioex Iogn Iohb Iotr
)

IF (BUILD_TESTING)
  ENABLE_TESTING()
ENDIF (BUILD_TESTING)
