  )

  NOX_TestCompare.C

  APPEND_SET(SOURCES NOX_Epetra_DebugTools.C)
ENDIF()

PACKAGE_ADD_LIBRARY(
  noxtestutils
  HEADERS ${HEADERS}
  SOURCES ${SOURCES}
  TESTONLY
  )
